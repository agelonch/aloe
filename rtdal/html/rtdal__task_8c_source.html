95 int rtdal_task_new_thread(pthread_t *thread, void *(*fnc)(void*), void *arg,
96  enum rtdal_task_type type, int prio, int cpu, int unblock_signals) {
97  hdebug("thread=0x%x,fnc=0x%x,arg=0x%x,type=%s,prio=%d,cpu=%d\n",thread,fnc,arg,
98  type==DETACHABLE?"detach":"join",prio,cpu);
99  RTDAL_ASSERT_PARAM(thread);
100  RTDAL_ASSERT_PARAM(fnc);
101  RTDAL_ASSERT_PARAM(prio>=0);
102  RTDAL_ASSERT_PARAM(cpu>=-1);
103 
104  pthread_attr_t attr;
105  int s,ret;
106  struct sched_param param;
107  cpu_set_t cpuset;
108  int i;
109 
110  ret = -1;
111 
112  s = pthread_attr_init(&attr);
113  if (s) {
114  RTDAL_POSERROR(s, "pthread_attr_init");
115  goto destroy_attr;
116  }
117 
118  s = pthread_attr_setinheritsched(&attr,PTHREAD_EXPLICIT_SCHED);
119  if (s) {
120  RTDAL_POSERROR(s,"pthread_attr_setinheritsched");
121  goto destroy_attr;
122  }
123 
124  s = pthread_attr_setdetachstate(&attr,
125  (type == DETACHABLE)?PTHREAD_CREATE_DETACHED:PTHREAD_CREATE_JOINABLE);
126  if (s) {
127  RTDAL_POSERROR(s, "pthread_attr_setdetachstate");
128  goto destroy_attr;
129  }
130 
131  if (cpu != -1) {
132  CPU_ZERO(&cpuset);
133  CPU_SET((size_t) cpu,&cpuset);
134  } else {
135  CPU_ZERO(&cpuset);
136  for (i=0;i<nof_cores;i++) {
137  CPU_SET((size_t) core_mapping[i],&cpuset);
138  }
139  }
140  s = pthread_attr_setaffinity_np(&attr, sizeof(cpu_set_t), &cpuset);
141  if (s) {
142  RTDAL_POSERROR(s, "pthread_attr_setaffinity_np");
143  goto destroy_attr;
144  }
145 
146  if (!getuid()) {
147  s = pthread_attr_setschedpolicy(&attr, (prio==TASK_DEFAULT_PRIORITY)?SCHED_RR:SCHED_FIFO);
148  if (s) {
149  if (s == EPERM) {
150  awarn("Not enough privileges to set task scheduling\n");
151  } else {
152  RTDAL_POSERROR(s,"pthread_attr_setschedpolicy");
153  goto destroy_attr;
154  }
155  }
156  }
157 
158  if (!getuid()) {
159  param.sched_priority = prio;
160  s = pthread_attr_setschedparam(&attr, &param);
161  if (s) {
162  if (s == EPERM) {
163  awarn("Not enough privileges to set task priority\n");
164  } else {
165  printf("prio=%d\n",prio);
166  RTDAL_POSERROR(s,"pthread_attr_setschedparam");
167  goto destroy_attr;
168  }
169  }
170  }
171 
172  s = pthread_create(thread, &attr, fnc, arg);
173  if (s) {
174  if (s == EPERM) {
175  awarn("Not enough privileges to set kernel thread priority\n");
176  } else {
177  RTDAL_POSERROR(s, "pthread_create");
178  goto destroy_attr;
179  }
180  }
181 
182  ret=0;
183 
184 destroy_attr:
185  pthread_attr_destroy(&attr);
186  return ret;
187 }
